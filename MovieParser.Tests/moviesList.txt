window.addEventListener('DOMContentLoaded', function() {
    var xhr;
    var sidebar = new StickySidebar('#movie-search-sidebar', {topSpacing: 10});

    function fetch(params) {
        if (xhr) xhr.abort();
        $('#movie-search-items').addClass('loading').load('/movie_list', params, function() {
            $('#movie-search-items').removeClass('loading');
            sidebar.updateSticky();
        });
        window.scroll({
            top: 0,
            behavior: 'smooth'
        });
    }

    $('#movie-search-params :input').change(function() {
        var params = $('#movie-search-params').serialize();
        window.history.pushState(null, null, '?'+params);
        fetch(params);
    });    

    $('#movie-search-items').on('click', '[data-page]', function() {
        var data = $(this).data();
        var params = $('#movie-search-params').serialize();
        if (params) {
            params += '&';
        }
        params += 'page='+data.page+'&since='+data.since;
        window.history.pushState(null, null, '?'+params);
        fetch(params);
    });

    $('#movie-search-items').on('click', '[data-tconst]', function() {
        var tconst = $(this).data('tconst');
        window.open('https://www.imdb.com/title/'+tconst+'/', '_blank');
        return false;
    });
});
